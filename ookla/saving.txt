      var names = getProviderNames(data);

      var chartHeight = 415;
      if (providerData.length >= 10) {
        chartHeight = 700
      } else if (providerData.length >= 6) {
        chartHeight = 600;
      }
      //- For columnrange charts, position the tooltip at the median point instead of the center.
      //- This is done by wrapping the tooltip's updatePosition function to:
      //-   1) figure out which line is being hovered,
      //-   2) compute where that line's median would be in relation to the range of the chart,
      //-   3) override the tooltip's horizontal position with the median position.
      //- The first step above is definitely the most brittle. Getting the index of the currently
      //- hovered line is done by getting the desired position of the tooltip and comparing it to
      //- each line's "tooltip position" object. If they match, assume this is the correct line.
      //- It's brittle because if the values are overridden at any point (similar to what we are
      //- doing here), no match will be found. Maybe there's a more graceful way to find this
      //- index, but it seems to work ok. Once you have the index, you can access that line's
      //- data, and the rest is more straightforward.
      //- This functionality is "nice-to-have" and can be removed if it causes future problems.
      Highcharts.wrap(Highcharts.Tooltip.prototype, 'updatePosition', function() {
        //- first arg is callback, second arg is a point
        var args = Array.prototype.slice.call(arguments, 1);
        //- only do this for columnrange charts
        if (this.chart.types.length === 1 && this.chart.types[0] === 'columnrange') {
          //- we need the index of the hovered line so we can do math on the data
          //- we can infer the index by comparing the arg's plotY to the series data tooltipPos[1]
          var tooltipPos = this.chart.series[0].data.map(function (data) {
            return data.tooltipPos[1];
          });
          var index = tooltipPos.indexOf(args[0].plotY);
          //- only modify the tooltip position if the line positions match up
          if (index > -1) {
            var median = this.chart.series[0].data[index].median; // the provider's median
            //- make sure the median value exists
            if (median) {
              //- columnrange charts are inverted by default, so y is horizontal
              var yAxis = this.chart.yAxis[0]; // yAxis for the whole chart
              //- compute the ratio of median to series range
              var percent = (median - yAxis.min) / (yAxis.max - yAxis.min);
              //- plotX is usually the center of the range, but it's overwritten here
              //- tooltip position doesn't care if the chart is inverted
              args[0].plotX = yAxis.width * percent; // map ratio to pixels
            }
          }
        }
        arguments[0].apply(this, args); // proceed
      });

      // render provider chart
      Highcharts.chart(providerId, {
        title: {
          text: `Expected speed range (download)`,

        xAxis: {

            },

              y: this.chart.plotTop + point.plotY - height - 10,

            color: 'white',

            }

        chart: {

          type: 'columnrange',
          inverted: true,
          marginBottom: 100,
          renderTo: providerId,
          //- (CT-5858) You must set the chart height or the chart will slowly get taller every time
          //- the viewport changes size, including when a mobile browser shows/hides chrome.
          height: chartHeight,
        },
        plotOptions: {
          series: {
            pointWidth: 3

          columnrange: {
            colorByPoint: true,
            dataLabels: {
              enabled: false
            },
          }
        },
        credits: {
          enabled: false,

        subtitle: {
          align: 'right',
          verticalAlign: 'bottom',
          floating: true,
          text: 'Oct 2023 - Dec 2023',
        },
        legend: {
          enabled: false,
        },
        series: [{
          name: names,
          data: providerData,
        }]
      });
    }

    $.each([ 'download', 'upload' ], function (metricIndex, metric) {
      var metricLabel = (metric === 'download') ? 'Download' : 'Upload';
      var id = 'performance-graph-historical-' + type + '-' + metric;
      var series = [];
      var lineChartColor = '#6afff3';
      var lineChartColorComparison = '#fff38e';
      if(metric === 'upload') {
        lineChartColor = '#bf71ff';
      }
      series.push({
        marker: marker,
        name: legendLabels.place[type][metric],
        data: transformData(type, metric, placeMonthlyData),

      });

      if (placeType === 'city' || placeType === 'postal-code') {

        series.push({
          marker: marker,
          name: legendLabels.region[type][metric],
          data: transformData(type, metric, regionMonthlyData),
          color: '#fff38e'
        });
      }

      //Locations Chart
      Highcharts.chart(id, {
        title: {
          text: `Historical performance (${metric})`,
        },
        tooltip: {
          dateTimeLabelFormats: {
            day: '%b %Y'
          }
        },
        chart: {
          type: 'spline',
          styledMode: true,
          marginBottom: 100,
          renderTo: id,
          //- (CT-5858) You must set the chart height or the chart will slowly get taller every time
          //- the viewport changes size, including when a mobile browser shows/hides chrome.
          height: 415,
        },
        subtitle: {
          align: 'right',
          verticalAlign: 'bottom',
          floating: true,
          text: 'Jan 2023 - Jan 2024',
        },
        credits: {
          enabled: false,
        },

          },
          accessibility: {
            rangeDescription: historicalRangeLabels.x,

        yAxis: {

            // escape single quotes since it's going in a js string
            rangeDescription: historicalRangeLabels.y[metric],
